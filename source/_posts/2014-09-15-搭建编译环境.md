## 安装samba ##
sudo apt-get install samba
sudo apt-get install smbfs
sudo apt-get install system-config-samba
## gcc降级 ##
sudo apt-get install gcc-4.4 g++-4.4 g++-4.4-multilib
cd /usr/bin
sudo mv gcc gcc.bak
sudo ln -s gcc-4.4 gcc
sudo mv g++ g++.bak
sudo ln -s g++-4.4 g++
